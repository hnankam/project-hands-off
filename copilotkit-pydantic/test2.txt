import math

def calculate_factorial_with_steps(n):
    if n < 0:
        return None, "Factorial is not defined for negative numbers."
    if n == 0:
        return 1, ["0! = 1 (by definition)"]

    current_result = 1
    steps_list = []
    
    # Initialize with 1! = 1, then proceed from 2
    steps_list.append(f"Starting with 1! = 1")

    for i in range(2, n + 1):
        previous_result = current_result
        current_result *= i
        steps_list.append(f"Step {i-1} ({i}!): {previous_result} * {i} = {current_result}")
