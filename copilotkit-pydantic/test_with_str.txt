def calculate_and_visualize_factorial(n):
    print(f"--- Running 'Computational Graph' for Factorial({n}) ---")
    print(f"Goal: Calculate {n}! (n factorial), which is {n} * ({n}-1) * ... * 1")
    print("-" * 60)

    factorial_value = 1
    intermediate_steps = []

    # Step 0: Initialize
    intermediate_steps.append(f"Initial value (representing 0! or empty product): {factorial_value}")

    # Loop to perform multiplications
    for i in range(1, n + 1):
        old_value = factorial_value
        factorial_value *= i
        if i == 1:
            intermediate_steps.append(f"Step {i}: {old_value} * {i} = {factorial_value} (This is 1!)")
        else:
            intermediate_steps.append(f"Step {i}: {old_value} * {i} = {factorial_value}")

    # Print the "graph" steps
    for step in intermediate_steps:
        print(step)

    print("-" * 60)
    print(f"Calculation Complete!")
    print(f"The factorial of {n} is: {factorial_value}")
    print("-" * 60)

    # Summarize results
    print("\n--- Summary of Factorial Calculation ---")
    print(f"Calculated Factorial of: {n}")
    print(f"Result ({n}!): {factorial_value}")
    print(f"This number has {len(str(factorial_value))} digits.")
    print("Factorial is a product of all positive integers less than or equal to n.")
    if factorial_value > 1_000_000_000_000:
        print(f"As you can see, {n}! is a very large number!")
    print("-" * 40)

    return factorial_value
